
- config.js字段说明
{
    "model": 1-搜索模式  2-推荐模式  3-混合模式
    "search":{      搜索模式配置
        "cycle":    执行搜索循环的次数，每一个循环会将下面的关键词都搜索一次
        "refresh":  每个关键词中，刷新视频的次数
        "like":     是否对当前出现的广告点赞或收藏（在未登陆账号时开启此功能会造成异常）
        "keywords": 搜索的关键词
        "interval": 每个视频的播放时间间隔
    }
    "recommend"{    推荐模式配置
        "refresh":  刷新视频的次数
        "like":     是否对当前出现的广告点赞或收藏（在未登陆账号时开启此功能会造成异常）
        "interval": 每个视频的播放时间间隔
    }
    "mix":{               混合模式配置
        "search_limit":   每次执行搜索词条的个数
    }
    "exit":"alt+q":       强制退出快捷键
}
